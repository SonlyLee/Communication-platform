通过数据动态生成导航菜单

Code

<cuf-nav triggered-event="click">
    <cuf-nav-item ng-repeat="item in navData" href="{{item.href}}" label="{{item.name}}" 
                       has-children="{{item.children && item.children.length !== 0}}">
            <cuf-nav-child-item ng-repeat="itemChild in item.children" href="{{itemChild.href}}" label="{{itemChild.name}}"></cuf-nav-child-item>
    </cuf-nav-item>
</cuf-nav>
